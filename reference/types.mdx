## `StartRunInput`

```ts
type StartRunInput = {
  name: string;
  userId?: string;
  customerId?: string;
  customerName?: string;
  customerPlan?: string | { name: string; monthlyUsd: number };
  properties?: Record<string, unknown>;
  budgetUsd?: number;
  startedAt?: string;
};
```

## `TrackCostInput`

```ts
type TrackCostInput = {
  runId: string;
  provider: string;
  label?: string;
  model?: string;
  inputTokens?: number;
  outputTokens?: number;
  cacheCreationInputTokens?: number;
  cacheCreationInputTokens5m?: number;
  cacheCreationInputTokens1h?: number;
  cachedInputTokens?: number;
  reasoningTokens?: number;
  costUsd?: number;
  latencyMs?: number;
  status?: string;
  metadata?: Record<string, unknown>;
  createdAt?: string;
};
```

## `TrackOutcomeInput`

```ts
type TrackOutcomeInput = {

};
```
